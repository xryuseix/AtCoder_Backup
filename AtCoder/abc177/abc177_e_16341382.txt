int ngcd(vector<int>& a) {
    int d = a[0];
    for (int i = 1; i < a.size() && d != 1; i++) d = gcd(a[i], d);
    return d;
}

vi w(1e6 + 10, 0);

void enum_div(int n) {
    for (int i = 2; i * i <= n; ++i) {
        if (n % i == 0) {
            // ret.push_back(i);
            w[i]++;
            // if (w[i] > 1) return 0;
            if (w[i] > 1) return;
            if (i != 1 && i * i != n) {
                // ret.push_back(n/i);
                w[n / i]++;
                if (n / i > 1 && w[n / i] > 1) return;
            }
        }
    }
    w[n]++;
    // if (w[n] > 1) return 0;
    // return 1;
}

int main() {
    int n;
    scanf("%d", &n);
    vi v(n);
    rep(i, n) scanf("%d", &v[i]);
    if (ngcd(v) != 1) {
        fin("not coprime");
    } else {
        bool ch = 1;
        rep(i, n) { enum_div(v[i]); }
        for (int i = 2; i < w.size(); i++) {
            if (w[i] > 1) {
                ch = 0;
                break;
            }
        }
        if (ch) {
            fin("pairwise coprime");
        } else {
            fin("setwise coprime");
        }
    }
}
