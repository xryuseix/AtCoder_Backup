bool judge(int diff, vi d, int n) {
    // cout<<"ここjudge1"<<endl;
    map<pair<int, int>, int> m;
    vi v;
    rep(i, d.size()) {
        m[ mp(d[i], 24 - d[i]) ]++;
        m[ mp(24 - d[i], d[i]) ]++;
        v.push_back(d[i]);
        v.push_back(24 - d[i]);
    }
    for(int i = 0; i < d.size(); i++) {
        int a = d[i];
        int b = d[i] + diff;
        B = b;
        if(b > 24) continue;
        if(judge2(a, b, v, m)) {
            return true;
        } else {}
    }
    return false;
}
