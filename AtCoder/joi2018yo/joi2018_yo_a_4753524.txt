#include<stdio.h>
int main(void){
	
	int n,a,b,c,d;
	scanf("%d %d %d %d %d",&n,&a,&b,&c,&d);
	if(n%a==0&&n%c==0){
		if((n/a)*b>(n/c)*d){
			printf("%d\n",(n/c)*d);
		}
		else
			printf("%d\n",(n/a)*b);
	}
	
	else if(n%a!=0&&n%c==0){
		if((n/a+1)*b>(n/c)*d){
			printf("%d\n",(n/c)*d);
		}
		else
			printf("%d\n",(n/a+1)*b);
	}
	else if(n%a==0&&n%c!=0){
		if((n/a)*b>(n/c+1)*d){
			printf("%d\n",(n/c+1)*d);
		}
		else
			printf("%d\n",(n/a)*b);
	}
	
	else if(n%a!=0&&n%c!=0){
		if((n/a+1)*b>(n/c+1)*d){
			printf("%d\n",(n/c+1)*d);
		}
		else
			printf("%d\n",(n/a+1)*b);
	}
	return 0;
}
