void solve(ld a, ld b, ld r) {
    ll num = 0;
    ll start = ceill(a - r);
    ll end = floorl(a + r);

    for (ld i = start; i <= end; i++) {
        ld p = sqrt(r * r - (a - i) * (a - i));

        ll bottom = ceill(b - p);
        ll top = floorl(b + p);
        num += top - bottom + 1;
    }
    cout << num << endl;
}

int main() {
    ld x, y, r;

    r = nextafter(r, INF);
    solve(x, y, r);
