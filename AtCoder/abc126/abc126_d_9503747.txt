void dfs(int n, int s){
    // cout<<"n;"<<n<<endl;
    rep(i,g[n].size()){
        if(used[g[n][i].first])continue;
        used[g[n][i].first]=1;
        if(g[n][i].second%2==0){
            ans[g[n][i].first]=s;
            dfs(g[n][i].first,s);
        }
        else{
            ans[g[n][i].first]=1-s;
            dfs(g[n][i].first,1-s);
        }
    }
}
