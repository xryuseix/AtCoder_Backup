void quicksort(ll a[],ll b[], ll first, ll last){
    ll i,j,x;

    x=a[(first+last)/2];
	i=first;
	j=last;
	while(1){
        while(a[i] < x) i++;
        while(x < a[j]) j--;
        if(i >= j) break;
        swap(a[i],a[j]);
        swap(b[i],b[j]);
        i++;
		j--;
    }
    if(first<i-1) quicksort(a,b,first,i-1);
    if(j+1<last) quicksort(a,b,j+1,last);
}
