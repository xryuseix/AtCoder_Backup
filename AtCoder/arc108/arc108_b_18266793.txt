    rep(i, n - 1) bf[i + 1] = i;
    int ans = 0;
    int p = 0;
    while (p + 2 < n) {
        string tmp = "";
        tmp += s[p];
        tmp += s[af[p]];
        tmp += s[af[af[p]]];
        // cout<<p<<" "<<tmp<<" "<<af[p]<<endl;
        if (tmp == "fox") {
            {
                s[p] = '-';
                s[af[p]] = '-';
                s[af[af[p]]] = '-';
            }
            ans++;
            int ne = af[af[af[p]]];
            if (ne == n) break;
            p = bf[p];
            if (p == n) {
                p = ne;
                bf[p] = n;
                continue;
            }
            af[p] = ne;
            bf[ne] = p;
            // cout<<ne<<" "<<bf[ne]<<" "<<af[ne]<<endl;
            if (bf[p] != n) {
                p = bf[p];
            }

        } else {
            p = af[p];
        }
    }
    // fin(s);
    // fin(ans);
    fin(n - ans * 3);
