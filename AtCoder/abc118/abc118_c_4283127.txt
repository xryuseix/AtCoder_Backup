int gcd(int x,int y){
		int t;
		while(y!=0){
			t=x%y;x=y;y=t;
		}
		return x;
}
int ngcd(int n,int a[]){
		int i,d;
		d=a[0];
		for(i=1;i<n&&d!=1;i++)d=gcd(a[i],d);
		return d;
}
