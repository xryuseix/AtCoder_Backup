void check(vvi &a, vi &v){
    ll t=0;
    rep(i,a.size()){
        if(v[a[i][1]] - v[a[i][0]] == a[i][2]){
            t+=a[i][3];
        }
    }
    chmax(ans,t);
}

void dfs(int pos, vi &v, int m, vvi &a) {
    if(pos==v.size()){
        check(a, v);
        return;
    }
    int fi=1;
    if(pos)fi=v[pos-1];
    for(int i=fi;i<=m;i++){
        v[pos]=i;
        dfs(pos+1, v, m, a);
    }
    return;
}
