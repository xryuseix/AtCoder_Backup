void umeru(string &s,vc &v,int n){
    if(v[0]=='S'){
        if(s[0]=='o'){
            v[n-1]=v[1];
        } else {
            if(v[1]=='W')v[n-1]='S';
            else v[n-1]='W';
        }
    } else{
        if(s[0]=='o'){
            if(v[1]=='W')v[n-1]='S';
            else v[n-1]='W';
        } else {
            v[n-1]=v[1];
        }
    }
    for(int i=1;i<n-2;i++){
        if(v[i]=='S'){
            if(s[i]=='o'){
                v[i+1]=v[i-1];
            } else {
                if(v[i-1]=='W')v[i+1]='S';
                else v[i+1]='W';
            }
        } else{
            if(s[i]=='o'){
                if(v[i-1]=='W')v[i+1]='S';
                else v[i+1]='W';
            } else {
                v[i+1]=v[i-1];
            }
        }
    }
}

bool judge(string &s,vc &v,int n){
    bool ch=true;
    rep(i,n){
        if(v[i]=='S'){
            if(s[i]=='o'){
                if(v[(i-1+n)%n]!=v[(i+1)%n])ch=false;
            } else {
                if(v[(i-1+n)%n]==v[(i+1)%n])ch=false;
            }
        } else {
            if(s[i]=='o'){
                if(v[(i-1+n)%n]==v[(i+1)%n])ch=false;
            } else {
                if(v[(i-1+n)%n]!=v[(i+1)%n])ch=false;
            }
        }
    }
    return ch;
}
