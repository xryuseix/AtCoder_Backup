int main() {
    ll n;
    cin >> n;
    vi v(3);
    while (n > 0) {
        v[n % 10 % 3]++;
        n /= 10;
    }
    if (v[0] == 0 && (v[1] == 0 || v[2] == 0))
        fin(-1);
    else {
        int ans = INF;
        if (v[1] > v[2]) swap(v[1], v[2]);
        {
            vi w = v;
            w[2] -= w[1];
            w[2] %= 3;
            chmin(ans, w[2]);
        }
        {
            vi w = v;
            w[1] %= 3;
            w[2] %= 3;
            if (w[1] > w[2]) swap(w[1], w[2]);
            w[2] -= w[1];
            chmin(ans, w[2]);
        }
        {
            vi w = v;
            w[1] %= 3;
            w[2] -= w[1];
            chmin(ans, w[2] % 3);
        }
        fin(ans);
    }
}
