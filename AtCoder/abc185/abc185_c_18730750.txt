ll nCk(ll n, ll r) {
    if (dp[n][r]) return dp[n][r];
    if (r == 0 || r == n)
        return (1);
    else if (r == 1)
        return (n);
    return dp[n][r] = (nCk(n - 1, r - 1) + nCk(n - 1, r));
}
