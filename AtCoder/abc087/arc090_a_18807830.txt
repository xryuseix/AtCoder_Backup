    rep(i, 2) rep(j, n) cin >> v[i][j];
    vvi dp(3, vi(n, 0));
    rep(i, 2) {
        rep(j, n) {
            int sum1 = v[i][j];
            int sum2 = v[i][j];
            if (i > 0) sum1 += dp[i - 1][j];
            if (j > 0) sum2 += dp[i][j - 1];
            chmax(dp[i][j], sum1);
            chmax(dp[i][j], sum2);
        }
    }
    fin(dp[1][n - 1]);
