        ll a,b;

        q.push(mp(b,a));
    }
    ll rest=n;
    ll ans=0;
    while(rest>0){
        auto p=q.top();q.pop();
        // cout<<p.first<<" "<<p.second<<endl;
        if(p.second <= rest){
            rest-=p.second;
            ans+=p.second*p.first;
        } else {
            ans+=rest*p.first;
            rest=0;
        }
    }
