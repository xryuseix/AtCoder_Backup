// vector vの中のn以下の数で最大のものを返す
ll bs(vector<ll> &v, ll x){
    ll ok = -1; //これがx以下 
    ll ng = v.size(); //x以上 
    // 問題によってokとngを入れ替える
    while(abs(ok - ng) > 1){
        ll mid = (ok + ng)/2;
        if(v[mid] <= x) ok = mid;
        else ng = mid;
    }
    return ok;
}
