int lis(vector<int>& v) {
    vi dp(1, v[0]);
    for(int i = 1; i < v.size(); i++) {
        if(v[i] > dp[dp.size() - 1]) {
            dp.push_back(v[i]);
        } else {
            int pos = distance(lower_bound(dp.begin(), dp.end(), v[i]), dp.begin());
            dp[pos] = v[i];
        }
    }
    return (int)dp.size();
}

int conlis(vector<int>& v) {
    vi dp(v.size() + 1, 0);
    int ans = 0;
    for(int i = 0; i < v.size(); i++) {
        dp[v[i]] = dp[v[i] - 1] + 1;
        ans = max(ans, dp[v[i]]);
    }
    return ans;
}
