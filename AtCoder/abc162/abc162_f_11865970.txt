int main() {
    
    ll n;
    cin>>n;
    vll v(n),maxNum(n);
    ll num=1,maxK=3;
    

    rep(i,n){
        scanf("%lld",&v[i]);
    }
    rep(i,n){
        if(i==maxK){
            num++;
            maxK+=2;
        }
        maxNum[i]=num;
    }
    
    vector<map<ll,ll>> dp(n,map<ll,ll>());
    for(int i=0;i<n;i++){
        dp[i][1]=v[i];
        for(int k=2;k<=4;k++){ //-4,-3,-2
            if(i-k<0)continue;
            for(auto l=dp[i-k].begin();l!=dp[i-k].end();l++){// kの場所のたて
                ll Val=l->second+v[i];
                ll kosu=l->first+1;
                if(kosu<maxNum[i]){
                    continue;
                } else {
                    if(dp[i].count(kosu)==1)chmax(dp[i][kosu],Val);
                    else dp[i][kosu]=Val;
                }
            }
        }
    }
    ll ans=-LLINF;
    for(int i=0;i<n;i++){
        irep(j,dp[i]){
            if(j->first==n/2){
                chmax(ans,j->second);
            }
        }
    }
    fin(ans);
