const ll LLINF = 1LL<<60;

vector<ll> enum_div(ll n){
    vector<ll> ret;
    for(ll i = 1 ; i*i <= n ; ++i){
        if(n%i == 0){
            ret.push_back(i);
            if(i != 1 && i*i != n){
                ret.push_back(n/i);
            }
        }
    }
    return ret;
}

int main(void) {
    
    ll n;
    cin>>n;
    vll v=enum_div(n);

    ll ans=LLINF;
    rep(i,v.size()){
        ll a=v[i];
        ll b=n/v[i];
        // cout<<a<<" "<<b<<endl;
        int dig1=0;
        int dig2=0;
        while(a>0){
            dig1++;
            a/=10;
        }
        while(b>0){
            dig2++;
            b/=10;
        }
        ans=min(ans,(ll)max(dig1,dig2));
    }
    fin(ans);

}
