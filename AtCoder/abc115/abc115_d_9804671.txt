ll rec(ll n, ll x){
    if(n==0)return 1;
    ll len=(1LL<<(n+1))-3; // レベルn-1のバーガーの長さ
    ll num=(1LL<<n)-1; // レベルn-1に含まれるP
    if(x==1)return 0;
    if(x<=len+1)return rec(n-1,x-1);
    if(x==len+2)return 1+num;
    else if(x<=2*(len+1)) return 1+num+rec(n-1,x-len-2);
    else return 2*num+1;
}
